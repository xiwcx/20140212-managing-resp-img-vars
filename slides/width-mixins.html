image width mixins

full width mixin

@mixin calc-proportions-cinematic( $width ) {
    margin-left: percentage( -(( 100 - $width ) / 2) / $width );
    width: percentage( 100 / $width );
}

custom width mixin

@mixin calc-proportions-inline( $width ) {
    margin-left: percentage((( 100 - $width ) / 2) / 100);
    width: percentage( $width / 100 );
}
